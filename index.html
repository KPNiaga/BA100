Gantikan URL Apps Script anda di sini (Gunakan URL baru selepas deploy tadi) const APPS_SCRIPT_URL = "https://script.google.com/macros/s/AKfycby_QzxRC7U9tEn0lsife7QxqGi1Ei8biCmJFlbK2WjFyeJAImTOKCd9DN2V7-BYwYsjAw/exec"; // 2. Sediakan pembolehubah kosong untuk simpan produk yang akan dimuat turun nanti let products = []; // 3. Fungsi untuk tarik data barang dari Google Sheet apabila laman web dibuka async function loadProductsFromSheet() { const container = document.getElementById("products-container"); container.innerHTML = `
Memuat turun senarai barangan terkini...
`; try { // Panggil Google Sheet API yang kita buat tadi const response = await fetch(`${APPS_SCRIPT_URL}?action=getProducts`); products = await response.json(); // Paparkan produk ke dalam borang HTML renderProducts(products); } catch (error) { console.error("Gagal memuat turun data barang:", error); container.innerHTML = `
Gagal memuat turun barangan. Sila segarkan (refresh) semula halaman.
`; } } // 4. Fungsi untuk paparkan produk ke HTML (Termasuk semakan Stock Availability) function renderProducts(productsList) { const container = document.getElementById("products-container"); container.innerHTML = ""; // Bersihkan loading state if (productsList.length === 0) { container.innerHTML = `
Tiada barangan ditemui.
`; return; } productsList.forEach(prod => { // Semak jika stok kosong const isOutOfStock = prod.stock <= 0; const card = document.createElement("div"); card.className = `p-4 rounded-xl border bg-white flex flex-col sm:flex-row items-center justify-between gap-4 transition-all ${isOutOfStock ? 'opacity-60 border-slate-200' : 'border-slate-100 hover:border-brand-200'}`; card.innerHTML = `
${prod.discount ? `${prod.discount} OFF` : ''}

${prod.name}

${isOutOfStock ? `Stok Habis` : `Baki Stok: ${prod.stock}` }
Karton (${prod.karton.size}) RM ${prod.karton.oldPrice.toFixed(2)} RM ${prod.karton.newPrice.toFixed(2)}
Unit (${prod.unit.size}) RM ${prod.unit.oldPrice.toFixed(2)} RM ${prod.unit.newPrice.toFixed(2)}
`; container.appendChild(card); }); } // 5. Jalankan fungsi load data secara automatik sejurus halaman siap dibuka window.addEventListener("DOMContentLoaded", loadProductsFromSheet);
